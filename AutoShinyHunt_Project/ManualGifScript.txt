 from PIL import Image, ImageTk
import tkinter as tk

class AnimatedGif(tk.Label):
    def __init__(self, master, gif_path, *args, **kwargs):
        super().__init__(master, *args, **kwargs)
        self.gif_path = gif_path
        self.frames = []
        self.current_frame = 0

        # Open the GIF and extract the frames
        with Image.open(gif_path) as im:
            for frame in range(im.n_frames):
                im.seek(frame)
                self.frames.append(ImageTk.PhotoImage(im.copy()))

        self.update_frame()

    def update_frame(self):
        # Update the label with the next frame
        self.config(image=self.frames[self.current_frame])
        self.current_frame = (self.current_frame + 1) % len(self.frames)
        self.after(100, self.update_frame)  # Refresh every 100ms for animation

# Tkinter setup
root = tk.Tk()
root.title("Animated GIF Example")

# Example usage with an animated GIF
gif_label = AnimatedGif(root, "C:/Users/googl/Documents/PythonScripts/PokeShinyHunt_v1/gen1_gifs/Aerodactyl.gif")
gif_label.pack()

root.mainloop()
